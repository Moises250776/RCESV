    <script>
        const audioPlayer = document.getElementById('audioPlayer');
        const playBtn = document.getElementById('playBtn');
        const radioLogo = document.getElementById('radioLogo');
        const metadataEl = document.getElementById('metadata');

        playBtn.addEventListener('click', () => {
            if (audioPlayer.paused) {
                audioPlayer.src = "https://stream.zeno.fm/3mqszxaw69duv";
                audioPlayer.play()
                    .then(() => {
                        playBtn.textContent = 'PAUSAR';
                        radioLogo.classList.add('playing');
                    })
                    .catch(err => console.error("Error al iniciar streaming:", err));
            } else {
                audioPlayer.pause();
                playBtn.textContent = 'REPRODUCIR';
                radioLogo.classList.remove('playing');
            }
        });

        function iniciarMetadata() {
            const urlMeta = "https://api.zeno.fm/mounts/metadata/subscribe/3mqszxaw69duv";
            const eventSource = new EventSource(urlMeta);

            eventSource.onmessage = function(event) {
                try {
                    const data = JSON.parse(event.data);
                    if (data && data.streamTitle) {
                        metadataEl.textContent = "Sonando: " + data.streamTitle;
                    } else {
                        metadataEl.textContent = "Radio Cristiana de El Salvador";
                    }
                } catch (e) {
                    metadataEl.textContent = "Radio Cristiana de El Salvador";
                }
            };

            eventSource.onerror = function() {
                metadataEl.textContent = "Radio Cristiana de El Salvador";
            };
        }

        iniciarMetadata();
    </script>
